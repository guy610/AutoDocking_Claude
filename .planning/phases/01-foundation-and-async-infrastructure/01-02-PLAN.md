class ProgressManager:
    """Manages progress display with 500ms delay per APP-02 requirement.

    Shows progress bar and cancel button only if operation exceeds 500ms.
    Supports both determinate (known total) and indeterminate (unknown duration) modes.
    """

    __init__(self, progress_bar: QProgressBar, cancel_button: QPushButton, status_bar: QStatusBar):
        - Store references to widgets
        - Create QTimer (single-shot, 500ms)
        - Connect timer.timeout to _show_widgets()
        - Initialize _is_running = False

    start(self):
        - Set _is_running = True
        - Start timer with 500ms delay
        - DO NOT show widgets yet (that's the point of the delay)

    _show_widgets(self):
        - Only if _is_running:
          - progress_bar.setVisible(True)
          - cancel_button.setVisible(True)

    update(self, current: int, total: int, message: str = ""):
        - If total > 0: Set progress_bar.setMaximum(total), setValue(current) [determinate]
        - If total == 0: Set progress_bar.setMinimum(0), setMaximum(0) [indeterminate]
        - If message: status_bar.showMessage(message)

    stop(self):
        - Set _is_running = False
        - Stop timer (in case operation finished before 500ms)
        - Hide progress_bar and cancel_button
        - Clear status bar message

def _demo_work(is_cancelled, progress_callback):
    """Demo operation that takes ~5 seconds and can be cancelled."""
    import time
    total = 100
    for i in range(total):
        if is_cancelled():
            return "Operation cancelled"
        progress_callback(i + 1, total, f"Processing item {i + 1} of {total}...")
        time.sleep(0.05)  # 50ms * 100 = 5 seconds total
    return "Demo operation completed successfully!"
